680 virtual QCString trWriteList(int numEntries)
681 {
682 QCString result;
683 int i;
684 // the inherits list contain `numEntries' classes
685 for (i=0;i<numEntries;i++)
686 {
687 // use generateMarker to generate placeholders for the class links!
688 result+=generateMarker(i); // generate marker for entry i in the list
689 // (order is left to right)
690
691 if (i!=numEntries-1) // not the last entry, so we need a separator
692 {
693 if (i<numEntries-2) // not the fore last entry
694 result+=", ";
695 else // the fore last entry
696 result+=" és ";
697 }
698 }
699 return result;
700 }
